114 size_t bind_executor(const executor& ex) {
115 if (!ex) throw std::invalid_argument("executor is nullptr");
116 #if ZCLIBLOG_LOGGER_CONFIGURATIONS_LOGGER_MUTEX
117 std::lock_guard<ZCLibLog_MUTEX> lock(m_mutex);
118 #endif
119 m_executors.emplace_back(m_nextID, ex);
120 return m_nextID++;
121 }
122
127 void debind_executor(size_t id) {
128 #if ZCLIBLOG_LOGGER_CONFIGURATIONS_LOGGER_MUTEX
129 std::lock_guard<ZCLibLog_MUTEX> lock(m_mutex);
130 #endif
131 m_executors.erase(
132 std::remove_if(
133 m_executors.begin(),
134 m_executors.end(),
135 [id](const executor_pair& p) { return p.first == id; }
136 ),
137 m_executors.end()
138 );
139 }
140
142 void clear_executors() {
143 #if ZCLIBLOG_LOGGER_CONFIGURATIONS_LOGGER_MUTEX
144 std::lock_guard<ZCLibLog_MUTEX> lock(m_mutex);
145 #endif
146 m_executors.clear();
147 m_nextID = {};
148 }
