23 struct boost_format : format_apis::traditional {
24 template <typename... Args>
25 static std::string do_format(FLogPack pack, const std::string& fmt, Args&&... args) {
26 std::string f_msg;
27 if (sizeof...(args) == 0) {
28 f_msg = fmt;
29 }
30 else {
31 boost::format f(fmt);
32 try {
33 apply(f, std::forward<Args>(args)...);
34 } catch (const std::exception& e) {
35 return e.what();
36 }
37 f_msg = f.str();
38 }
39
40 const auto tp = std::chrono::system_clock::time_point(std::chrono::milliseconds(pack.time));
41
42 const char* level_str;
43 switch (pack.level) {
44 case LogLevel::TRACE: level_str = "[TRACE]";
45 break;
46 case LogLevel::DEBUG: level_str = "[DEBUG]";
47 break;
48 case LogLevel::INFO: level_str = "[INFO]";
49 break;
50 case LogLevel::WARN: level_str = "[WARN]";
51 break;
52 case LogLevel::ERROR: level_str = "[ERROR]";
53 break;
54 case LogLevel::FATAL: level_str = "[FATAL]";
55 break;
56 default: level_str = "[OUT]";
57 break;
58 }
59
60 std::time_t tt = std::chrono::system_clock::to_time_t(tp);
61 std::tm tm{};
62
63 #if defined(_WIN32)
64 localtime_s(&tm, &tt);
65 #elif defined(__linux__) || defined(__APPLE__) || defined(__unix__)
66 localtime_r(&tt, &tm);
67 #else
68 tm = *std::localtime(&tt);
69 #endif
70
71 std::ostringstream t_oss;
72 t_oss << std::put_time(&tm, "%Y-%m-%d %H:%M:%S");
73
74 return (boost::format("%1% [%2%] %3% %4%")
75 % t_oss.str()
76 % *pack.name
77 % level_str
78 % f_msg
79 ).str();
80 }
81 private:
82 template <typename T>
83 static void apply(boost::format& f, T&& t) {
84 f % std::forward<T>(t);
85 }
86
87 template <typename T, typename... Args>
88 static void apply(boost::format& f, T&& t, Args&&... args) {
89 f % std::forward<T>(t);
90 apply(f, std::forward<Args>(args)...);
91 }
92 };
93 }
