91 void execute(const std::string& message, const LogLevel level) const {
92 #if ZCLIBLOG_LOGGER_CONFIGURATIONS_LOGGER_SYNC_MUTEX
93 std::lock_guard<ZCLibLog_MUTEX> lock(m_mutex);
94 #endif
95 if (!message.empty()) {
96 // ReSharper disable once CppUseElementsView
97 // ReSharper disable once CppUseStructuredBinding
98 for (const auto& the_executor_pair : m_executors) {
99 the_executor_pair.second->do_execute(message, level);
100 }
101 }
102 }

109 ZCLibLog_NODISCARD bool be_executable(const LogLevel level) const noexcept {
110 return m_config.min_level <= level && level <= m_config.max_level;
111 }
112
114 ZCLibLog_NODISCARD bool has_executor() const {
115 #if ZCLIBLOG_LOGGER_CONFIGURATIONS_LOGGER_SYNC_MUTEX
116 std::lock_guard<ZCLibLog_MUTEX> lock(m_mutex);
117 #endif
118 return !m_executors.empty();
119 }
120
126 size_t bind_executor(const executor& ex) {
127 if (!ex) throw std::invalid_argument("executor is nullptr");
128 #if ZCLIBLOG_LOGGER_CONFIGURATIONS_LOGGER_SYNC_MUTEX
129 std::lock_guard<ZCLibLog_MUTEX> lock(m_mutex);
130 #endif
131 m_executors.emplace_back(m_nextID, ex);
132 return m_nextID++;
133 }
134
139 void debind_executor(size_t id) {
140 #if ZCLIBLOG_LOGGER_CONFIGURATIONS_LOGGER_SYNC_MUTEX
141 std::lock_guard<ZCLibLog_MUTEX> lock(m_mutex);
142 #endif
143 m_executors.erase(
144 std::remove_if(
145 m_executors.begin(),
146 m_executors.end(),
147 [id](const executor_pair& p) { return p.first == id; }
148 ),
149 m_executors.end()
150 );
151 }
152
154 void clear_executors() {
155 #if ZCLIBLOG_LOGGER_CONFIGURATIONS_LOGGER_SYNC_MUTEX
156 std::lock_guard<ZCLibLog_MUTEX> lock(m_mutex);
157 #endif
158 m_executors.clear();
159 m_nextID = {};
160 }

186 ZCLibLog_NODISCARD LogPack get_log_pack() const {
187 const auto now = std::chrono::system_clock::now();
188 const auto ms = std::chrono::duration_cast<std::chrono::milliseconds>(
189 now.time_since_epoch()
190 ).count();
191
192 LogPack p;
193 p.name = &m_logger->name();
194 p.level = m_level;

201 ZCLibLog_NODISCARD bool check_executable() const {
202 if (

206 return {};
207 }

232 template <typename Fmt, typename... Args>
233 #ifdef ZCLibLog_HAS_CONSTRAINTS
234 requires is_format_api<Formatter, format_apis::traditional>::value

236 void operator()(Fmt&& fmt, Args&&... args) const {
237 if (!check_executable()) return;
238 const std::string Formatted = Formatter::do_format(
239 get_log_pack(),

241 std::forward<Args&&>(args)...
