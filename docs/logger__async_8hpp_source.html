43 class ThreadPool {
44 public:
49 explicit ThreadPool(const size_t numThreads = 1) : stop(false) {
50 for (size_t i = 0; i < numThreads; ++i) {
51 workers.emplace_back(
52 [this] {
53 while (true) {
54 std::function<void()> task;
55 {
56 std::unique_lock<std::mutex> lock(mtx);
57 cv.wait(lock, [this] { return stop || !tasks.empty(); });
58 if (stop && tasks.empty()) return;
59 task = std::move(tasks.front());
60 tasks.pop();
61 }
62 task();
63 }
64 }
65 );
66 }
67 }
68
70 ~ThreadPool() {
71 {
72 std::lock_guard<std::mutex> lock(mtx);
73 stop = true;
74 }
75 cv.notify_all();
76 for (auto& t : workers) t.join();
77 }
78
83 void submit(std::function<void()> task) {
84 {
85 std::lock_guard<std::mutex> lock(mtx);
86 tasks.push(std::move(task));
87 }
88 cv.notify_one();
89 }
90
91 private:
92 std::vector<std::thread> workers;
93 std::queue<std::function<void()>> tasks;
94 std::mutex mtx;
95 std::condition_variable cv;
96 std::atomic<bool> stop;
97 };
98 }
