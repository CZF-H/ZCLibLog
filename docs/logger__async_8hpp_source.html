64 class ThreadPool {
65 public:
70 explicit ThreadPool(const size_t numThreads = 1) : stop(false) {
71 for (size_t i = 0; i < numThreads; ++i) {
72 workers.emplace_back([this] {
73 while (true) {
74 std::function<void()> task;
75 {
76 std::unique_lock<std::mutex> lock(mtx);
77 cv.wait(lock, [this] { return stop || !tasks.empty(); });
78 if (stop && tasks.empty()) return;
79 task = std::move(tasks.front());
80 tasks.pop();
81 }
82 task();
83 }
84 });
85 }
86 }
87
89 ~ThreadPool() {
90 {
91 std::lock_guard<std::mutex> lock(mtx);
92 stop = true;
93 }
94 cv.notify_all();
95 for (auto& t : workers) t.join();
96 }
97
102 void submit(std::function<void()> task) {
103 {
104 std::lock_guard<std::mutex> lock(mtx);
105 tasks.push(std::move(task));
106 }
107 cv.notify_one();
108 }
109
110 private:
111 std::vector<std::thread> workers;
112 std::queue<std::function<void()>> tasks;
113 std::mutex mtx;
114 std::condition_variable cv;
115 std::atomic<bool> stop;
116 };
117 }

166 void execute(const std::shared_ptr<std::string>& message, const LogLevel level) const {
167 if (!message->empty()) {
168 LoggerAsync_ThreadPool.submit(
169 [this, message, level] {
170 #if ZCLIBLOG_LOGGER_CONFIGURATIONS_LOGGER_ASYNC_MUTEX
171 std::lock_guard<ZCLibLog_MUTEX> lock(m_mutex);
172 #endif
173 // ReSharper disable once CppUseStructuredBinding
174 // ReSharper disable once CppUseElementsView
175 for (const auto& the_executor_pair : m_executors) {
176 the_executor_pair.second->do_execute(*message, level);
177 }
178 }
179 );
180 }
181 }
182
188 ZCLibLog_NODISCARD bool be_executable(const LogLevel level) const noexcept {
189 return m_config.min_level <= level && level <= m_config.max_level;
190 }
191
193 ZCLibLog_NODISCARD bool has_executor() const {
194 #if ZCLIBLOG_LOGGER_CONFIGURATIONS_LOGGER_ASYNC_MUTEX
195 std::lock_guard<ZCLibLog_MUTEX> lock(m_mutex);
196 #endif
197 return !m_executors.empty();
198 }

205 size_t bind_executor(const executor& ex) {
206 if (!ex) throw std::invalid_argument("executor is nullptr");
207 #if ZCLIBLOG_LOGGER_CONFIGURATIONS_LOGGER_ASYNC_MUTEX
208 std::lock_guard<ZCLibLog_MUTEX> lock(m_mutex);
209 #endif
210 m_executors.emplace_back(m_nextID, ex);
211 return m_nextID++;
212 }
213
218 void debind_executor(size_t id) {
219 #if ZCLIBLOG_LOGGER_CONFIGURATIONS_LOGGER_ASYNC_MUTEX
220 std::lock_guard<ZCLibLog_MUTEX> lock(m_mutex);
221 #endif
222 m_executors.erase(
223 std::remove_if(
224 m_executors.begin(),
225 m_executors.end(),
226 [id](const executor_pair& p) { return p.first == id; }
227 ),
228 m_executors.end()
229 );
230 }
231
233 void clear_executors() {
234 #if ZCLIBLOG_LOGGER_CONFIGURATIONS_LOGGER_ASYNC_MUTEX
235 std::lock_guard<ZCLibLog_MUTEX> lock(m_mutex);
236 #endif
237 m_executors.clear();
238 m_nextID = {};
239 }
