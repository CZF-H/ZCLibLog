33 const auto tp = std::chrono::system_clock::time_point(std::chrono::milliseconds(pack.time));
34
35 const char* level_str;

37 case LogLevel::TRACE: level_str = "[TRACE]";

39 case LogLevel::DEBUG: level_str = "[DEBUG]";
40 break;
41 case LogLevel::INFO: level_str = "[INFO]";
42 break;
43 case LogLevel::WARN: level_str = "[WARN]";
44 break;
45 case LogLevel::ERROR: level_str = "[ERROR]";
46 break;
47 case LogLevel::FATAL: level_str = "[FATAL]";
48 break;
49 default: level_str = "[OUT]";
50 break;
51 }
52
53 return std::format(
54 "{:%Y-%m-%d %H:%M:%S} [{}] {} {}",
55 std::chrono::time_point_cast<std::chrono::milliseconds>(tp),
56 *pack.name,
